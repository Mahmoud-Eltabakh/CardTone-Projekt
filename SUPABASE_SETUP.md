# Supabase Setup for CardTone


Run the following SQL in your Supabase SQL Editor to create the necessary tables and policies.

```sql
-- Enable UUID extension (optional, but good to have)
create extension if not exists "uuid-ossp";

-- 1. PARENTS TABLE
create table parents (
  id text primary key,
  name text,
  kids jsonb default '[]'::jsonb, -- Array of kid IDs
  created_at timestamptz default now()
);

-- 2. KIDS TABLE
create table kids (
  id text primary key,
  name text,
  profile_picture_url text,
  kiosk_locked boolean default false,
  kiosk_pin text,
  master_card_id text,
  linked_parents jsonb default '[]'::jsonb, -- Array of parent IDs
  created_at timestamptz default now()
);

-- 3. KID SETTINGS
create table kid_settings (
  kid_id text primary key references kids(id) on delete cascade,
  config jsonb default '{}'::jsonb -- Stores maxVolume, playbackTimer, etc.
);

-- 4. CARDS
create table cards (
  id text primary key, -- This is the unique ID for the card record (often same as cardId)
  kid_id text references kids(id) on delete cascade,
  title text,
  type text default 'audio', -- 'audio', 'video', 'photo', 'master_kiosk', 'session_lock'
  uri text,
  cardId text, -- The physical NFC tag ID (e.g., '04:A3:...')
  created_at timestamptz default now()
);

-- 5. ACTIVITY LOGS
create table activity_logs (
  id bigint generated by default as identity primary key,
  kid_id text references kids(id) on delete cascade,
  type text,
  description text,
  metadata jsonb default '{}'::jsonb,
  created_at timestamptz default now()
);

-- 6. COMMANDS
create table commands (
  id bigint generated by default as identity primary key,
  kid_id text references kids(id) on delete cascade,
  command text,
  executed boolean default false,
  created_at timestamptz default now()
);

-- 7. PLAYBACK STATE
create table playback_state (
  kid_id text primary key references kids(id) on delete cascade,
  is_playing boolean default false,
  current_card_id text,
  current_track_title text,
  position int,
  duration int,
  last_update timestamptz default now()
);

-- Enable Realtime for all tables
alter publication supabase_realtime add table parents;
alter publication supabase_realtime add table kids;
alter publication supabase_realtime add table kid_settings;
alter publication supabase_realtime add table cards;
alter publication supabase_realtime add table activity_logs;
alter publication supabase_realtime add table commands;
alter publication supabase_realtime add table playback_state;

-- RLS Policies (Open Access for Development)
-- IMPORTANT: In production, you must restrict these policies!
alter table parents enable row level security;
create policy "Public access" on parents for all using (true);

alter table kids enable row level security;
create policy "Public access" on kids for all using (true);

alter table kid_settings enable row level security;
create policy "Public access" on kid_settings for all using (true);

alter table cards enable row level security;
create policy "Public access" on cards for all using (true);

alter table activity_logs enable row level security;
create policy "Public access" on activity_logs for all using (true);

alter table commands enable row level security;
create policy "Public access" on commands for all using (true);

alter table playback_state enable row level security;
create policy "Public access" on playback_state for all using (true);
```
