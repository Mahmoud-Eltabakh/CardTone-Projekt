# Supabase Setup — CardTone

> **Already ran the old schema?** Skip to [Migration from Old Schema](#migration-from-old-schema) at the bottom.

---

## Architecture

| Actor | Auth type | `auth.uid()` |
| --- | --- | --- |
| Parent app | OTP Magic Link (email) | Parent's Supabase user ID |
| Kid app | Anonymous sign-in (`signInAnonymously`) | Kid's Supabase user ID |

All RLS policies are written in terms of `auth.uid()`. The kid's database row `id` equals their anonymous `auth.uid()`. The parent's row `id` equals their OTP `auth.uid()`.

`kids.parents` is a JSONB array of parent UUIDs. A `"pending:uuid"` prefix means the link is awaiting approval from an existing parent. Linking and unlinking **only** happen through the `SECURITY DEFINER` RPC functions below — not via direct table writes.

## Fresh Setup

Run the entire block below in **Supabase → SQL Editor**:

```sql
-- ── Extensions ────────────────────────────────────────────────────────────────
create extension if not exists "uuid-ossp";

-- ── 1. PARENTS ────────────────────────────────────────────────────────────────
create table parents (
  id          text        primary key,           -- OTP auth.uid()
  name        text,
  kids        jsonb       not null default '[]', -- array of linked kid IDs
  created_at  timestamptz not null default now()
);

-- ── 2. KIDS ───────────────────────────────────────────────────────────────────
create table kids (
  id                          text        primary key, -- anonymous auth.uid()
  name                        text,
  parents                     jsonb       not null default '[]', -- ["uuid"] or ["pending:uuid"]
  kiosk_locked                boolean     not null default false,
  session_locked              boolean     not null default false,
  master_card_kiosk_mode_id   text,
  master_card_session_lock_id text,
  bedtime_start               text        not null default '',   -- "HH:MM", empty = off
  bedtime_end                 text        not null default '',   -- "HH:MM", empty = off
  session_timeout             int         not null default 0
    constraint session_timeout_non_negative check (session_timeout >= 0),
  session_started_at          timestamptz,
  max_volume                  int         not null default 100
    constraint max_volume_range check (max_volume between 0 and 100),
  created_at                  timestamptz not null default now()
);

-- ── 3. CARDS ──────────────────────────────────────────────────────────────────
create table cards (
  id          text        primary key,            -- NFC UID
  kid_id      text        not null references kids(id) on delete cascade,
  card_id     text,                               -- NFC UID duplicate (read by kid app)
  title       text        not null default '',
  type        text        not null default 'audio'
    constraint valid_card_type
    check (type in ('audio','video','youtube','photo','webview')),
  uri         text,
  created_at  timestamptz not null default now(),
  unique (kid_id, card_id)
);

-- ── 4. ACTIVITY LOGS ──────────────────────────────────────────────────────────
create table activity_logs (
  id          bigint      generated by default as identity primary key,
  kid_id      text        not null references kids(id) on delete cascade,
  type        text        not null,
  description text        not null default '',
  metadata    jsonb       not null default '{}',
  created_at  timestamptz not null default now()
);

-- ── 5. COMMANDS ───────────────────────────────────────────────────────────────
create table commands (
  id          bigint      generated by default as identity primary key,
  kid_id      text        not null references kids(id) on delete cascade,
  command     text        not null,
  executed    boolean     not null default false,
  created_at  timestamptz not null default now()
);

-- ── 6. PLAYBACK STATE ─────────────────────────────────────────────────────────
create table playback_state (
  kid_id              text        primary key references kids(id) on delete cascade,
  is_playing          boolean     not null default false,
  current_card_id     text,
  current_track_title text,
  position            int,
  duration            int,
  last_update         timestamptz not null default now()
);

-- ── Indexes ───────────────────────────────────────────────────────────────────
-- GIN indexes power the @> containment checks in all RLS policies.
create index idx_kids_parents_gin    on kids    using gin (parents);
create index idx_parents_kids_gin    on parents using gin (kids);

-- Partial index for fast pending-command lookups (stream filter).
create index idx_commands_pending    on commands (kid_id, created_at asc)
  where executed = false;

-- Time-ordered activity log queries.
create index idx_activity_logs_time  on activity_logs (kid_id, created_at desc);

-- ── Realtime ──────────────────────────────────────────────────────────────────
alter publication supabase_realtime add table parents;
alter publication supabase_realtime add table kids;
alter publication supabase_realtime add table cards;
alter publication supabase_realtime add table activity_logs;
alter publication supabase_realtime add table commands;
alter publication supabase_realtime add table playback_state;

-- ── Row Level Security ────────────────────────────────────────────────────────

-- PARENTS ──────────────────────────────────────────────────────────────────────
alter table parents enable row level security;

-- Parent reads and manages their own row.
create policy "parents: own row"
  on parents for all
  using  (auth.uid()::text = id)
  with check (auth.uid()::text = id);

-- Kid app reads names of parents it is linked to (for the PairingScreen display).
create policy "kids: read linked parent names"
  on parents for select
  using (
    exists (
      select 1 from kids
      where kids.id = auth.uid()::text
        and kids.parents @> jsonb_build_array(parents.id)
    )
  );

-- KIDS ─────────────────────────────────────────────────────────────────────────
alter table kids enable row level security;

-- Kid device has full access to its own row.
create policy "kids: own row"
  on kids for all
  using  (auth.uid()::text = id)
  with check (auth.uid()::text = id);

-- Linked parent can read a kid's row.
create policy "parents: read linked kids"
  on kids for select
  using (parents @> jsonb_build_array(auth.uid()::text));

-- Linked parent can update a kid's settings.
-- The WITH CHECK ensures a parent cannot remove themselves via a direct UPDATE.
create policy "parents: update linked kids"
  on kids for update
  using  (parents @> jsonb_build_array(auth.uid()::text))
  with check (parents @> jsonb_build_array(auth.uid()::text));

-- CARDS ────────────────────────────────────────────────────────────────────────
alter table cards enable row level security;

-- Kid device manages its own cards.
create policy "kids: own cards"
  on cards for all
  using  (kid_id = auth.uid()::text)
  with check (kid_id = auth.uid()::text);

-- Linked parent manages cards for their linked kids.
create policy "parents: linked kid cards"
  on cards for all
  using (
    exists (
      select 1 from kids
      where kids.id = cards.kid_id
        and kids.parents @> jsonb_build_array(auth.uid()::text)
    )
  )
  with check (
    exists (
      select 1 from kids
      where kids.id = cards.kid_id
        and kids.parents @> jsonb_build_array(auth.uid()::text)
    )
  );

-- ACTIVITY LOGS ────────────────────────────────────────────────────────────────
alter table activity_logs enable row level security;

create policy "kids: own logs insert"
  on activity_logs for insert
  with check (kid_id = auth.uid()::text);

create policy "kids: own logs select"
  on activity_logs for select
  using (kid_id = auth.uid()::text);

-- Linked parent can read and delete logs for their linked kids.
create policy "parents: linked kid logs"

  using (

-- COMMANDS ─────────────────────────────────────────────────────────────────────

create policy "kids: update own commands"
  on commands for update

create policy "kids: own playback state"

  using  (kid_id = auth.uid()::text)

-- ── RPC Functions (SECURITY DEFINER) ─────────────────────────────────────────
-- These are the ONLY way to modify kids.parents or parents.kids.
-- They include their own authorization checks and bypass RLS to avoid the
-- chicken-and-egg problem of needing to be linked before you can link.

-- request_link_kid: called by parent app when scanning a kid's QR code.
create or replace function request_link_kid(p_kid_id text, p_parent_id text)
returns text
language plpgsql
security definer
set search_path = public
as $$
declare
  v_parents      jsonb;
  v_real_parents text[];
begin
  if auth.uid()::text != p_parent_id then
    raise exception 'Unauthorized';
  end if;

  select parents into v_parents from kids where id = p_kid_id;
  if not found then
    insert into kids (id, parents) values (p_kid_id, '[]');
    v_parents := '[]';
  end if;

  select array_agg(elem)
  into   v_real_parents
  from   jsonb_array_elements_text(v_parents) as elem
  where  elem not like 'pending:%';

  if p_parent_id = any(coalesce(v_real_parents, '{}')) then
    update parents
      set kids = (
        select jsonb_agg(distinct e)
        from   jsonb_array_elements_text(kids || jsonb_build_array(p_kid_id)) as e
      )
      where id = p_parent_id;
    return 'linked';
  end if;

  if v_real_parents is null or array_length(v_real_parents, 1) is null then
    update kids
      set parents = (
        select jsonb_agg(distinct e)
        from   jsonb_array_elements_text(
                 v_parents || jsonb_build_array(p_parent_id)
               ) as e
        where  e != 'pending:' || p_parent_id
      )
      where id = p_kid_id;

    update parents
      set kids = (
        select jsonb_agg(distinct e)
        from   jsonb_array_elements_text(kids || jsonb_build_array(p_kid_id)) as e
      )
      where id = p_parent_id;

    return 'linked';
  else
    if not (v_parents @> jsonb_build_array('pending:' || p_parent_id)) then
      update kids
        set parents = v_parents || jsonb_build_array('pending:' || p_parent_id)
        where id = p_kid_id;
    end if;
    return 'pending';
  end if;
end;
$$;

-- unlink_kid: either the kid or the parent being removed may call this.
create or replace function unlink_kid(p_kid_id text, p_parent_id text)
returns void
language plpgsql
security definer
set search_path = public
as $$
begin
  if auth.uid()::text != p_kid_id and auth.uid()::text != p_parent_id then
    raise exception 'Unauthorized';
  end if;

  update kids
    set parents = coalesce(
      (
        select jsonb_agg(e)
        from   jsonb_array_elements_text(parents) as e
        where  e != p_parent_id
           and e != 'pending:' || p_parent_id
      ),
      '[]'
    )
    where id = p_kid_id;

  update parents
    set kids = coalesce(
      (
        select jsonb_agg(e)
        from   jsonb_array_elements_text(kids) as e
        where  e != p_kid_id
      ),
      '[]'
    )
    where id = p_parent_id;
end;
$$;

-- approve_pending_parent: called by an already-linked parent to admit a pending one.
create or replace function approve_pending_parent(p_kid_id text, p_parent_id_to_approve text)
returns void
language plpgsql
security definer
set search_path = public
as $$
declare
  v_caller_linked boolean;
begin
  select (parents @> jsonb_build_array(auth.uid()::text))
  into   v_caller_linked
  from   kids where id = p_kid_id;

  if not found          then raise exception 'Kid not found';  end if;
  if not v_caller_linked then raise exception 'Unauthorized'; end if;

  update kids
    set parents = (
      select coalesce(jsonb_agg(
        case when e = 'pending:' || p_parent_id_to_approve
             then p_parent_id_to_approve else e end
      ), '[]')
      from jsonb_array_elements_text(parents) as e
    )
    where id = p_kid_id;

  update parents
    set kids = (
      select jsonb_agg(distinct e)
      from   jsonb_array_elements_text(kids || jsonb_build_array(p_kid_id)) as e
    )
    where id = p_parent_id_to_approve;
end;
$$;

-- reject_pending_parent: called by an already-linked parent to deny a pending one.
create or replace function reject_pending_parent(p_kid_id text, p_parent_id_to_reject text)
returns void
language plpgsql
security definer
set search_path = public
as $$
declare
  v_caller_linked boolean;
begin
  select (parents @> jsonb_build_array(auth.uid()::text))
  into   v_caller_linked
  from   kids where id = p_kid_id;

  if not found          then raise exception 'Kid not found';  end if;
  if not v_caller_linked then raise exception 'Unauthorized'; end if;

  update kids
    set parents = coalesce(
      (
        select jsonb_agg(e)
        from   jsonb_array_elements_text(parents) as e
        where  e != 'pending:' || p_parent_id_to_reject
      ),
      '[]'
    )
    where id = p_kid_id;
end;
$$;

-- Grant RPC access to all authenticated users (OTP parents and anonymous kids).
grant execute on function request_link_kid           to authenticated;
grant execute on function unlink_kid                 to authenticated;
grant execute on function approve_pending_parent     to authenticated;
grant execute on function reject_pending_parent      to authenticated;
```

## Migration from Old Schema

> **Breaking change:** Kid IDs are now Supabase anonymous auth UIDs (UUID format).
> Existing `kid_TIMESTAMP` IDs will no longer be used. Existing parent–kid pairings must be
> re-established after the update by scanning the kid's QR code again from the parent app.

Run the following in **Supabase → SQL Editor**:

```sql
-- 1. Drop all open "Public access" policies
drop policy if exists "Public access" on parents;
drop policy if exists "Public access" on kids;
drop policy if exists "Public access" on cards;
drop policy if exists "Public access" on activity_logs;
drop policy if exists "Public access" on commands;
drop policy if exists "Public access" on playback_state;
drop policy if exists "Public access" on kid_settings;

-- 2. Rename linked_parents → parents (if you ran the very first schema version)
alter table kids rename column if exists linked_parents to parents;

-- 3. Add missing columns to kids
alter table kids add column if not exists session_locked              boolean not null default false;
alter table kids add column if not exists master_card_kiosk_mode_id  text;
alter table kids add column if not exists master_card_session_lock_id text;
alter table kids add column if not exists bedtime_start              text    not null default '';
alter table kids add column if not exists bedtime_end                text    not null default '';
alter table kids add column if not exists session_timeout            int     not null default 0;
alter table kids add column if not exists max_volume                 int     not null default 100;

-- 4. Add CHECK constraints
alter table kids
  add constraint if not exists session_timeout_non_negative check (session_timeout >= 0),
  add constraint if not exists max_volume_range check (max_volume between 0 and 100);

alter table cards
  add constraint if not exists valid_card_type
    check (type in ('audio','video','youtube','photo','webview'));

-- 5. Rename cardId → card_id on cards (camelCase → snake_case)
alter table cards rename column if exists "cardId" to card_id;

-- 6. Drop obsolete columns
alter table kids drop column if exists kiosk_pin;
alter table kids drop column if exists master_card_id;
alter table kids drop column if exists profile_picture_url;
drop table  if exists kid_settings;

-- 7. Add GIN indexes for RLS policy performance
create index if not exists idx_kids_parents_gin   on kids    using gin (parents);
create index if not exists idx_parents_kids_gin   on parents using gin (kids);
create index if not exists idx_commands_pending   on commands (kid_id, created_at asc)
  where executed = false;
create index if not exists idx_activity_logs_time on activity_logs (kid_id, created_at desc);
```
